Servant

  • Type Level APIs
  • Haskell API wrapper

             
  type API = -- GET /api/image?red=_&green=_&blue=_&imageId=_&handwritingId=_&text=_
             "api" :> "image"        :> QueryParam "red"           Word8
                                     :> QueryParam "green"         Word8
                                     :> QueryParam "blue"          Word8
                                     :> QueryParam "imageId"       String
                                     :> QueryParam "handwritingId" String
                                     :> QueryParam "text"          String
                                     :> Get '[OctetStream]         BS.ByteString
                                     
             -- GET /api/handwritings
        :<|> "api" :> "handwritings" :> Get '[JSON] [Handwriting]
         
             -- POST /api/sendImage?imageId=_&phoneNumber=_
        :<|> "api" :> "sendImage"    :> QueryParam "imageId"     String
                                     :> QueryParam "phoneNumber" String
                                     :> Post '[JSON]             ()
        :<|> Raw

AngularJS


// app.js
angular
  .module('frontendApp', [
    'ngAnimate',
    'ngCookies',
    'ngResource',
    'ngRoute',
    'ngSanitize',
    'ngTouch',
    'ui.materialize',
    'ngAnimate'
  ])
  .config(function ($routeProvider) {
    $routeProvider
      .when('/', {
        templateUrl: 'views/main.html',
        controller: 'MainCtrl',
        controllerAs: 'main'
      })
      .when('/about', {
        templateUrl: 'views/about.html',
        controller: 'AboutCtrl',
        controllerAs: 'about'
      })
      .otherwise({
        redirectTo: '/'
      });
  });
